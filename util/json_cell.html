import inspect
from dataclasses import dataclass
from typing import Any, Mapping, Optional
from pathlib import Path

from fnpcell import all as fp


@fp.pcell_class()
@dataclass(eq=False)
class JsonCell(fp.PCell):
    json_folder: fp.StrPath = fp.Param().as_field()
    json_name: fp.StrPath = fp.Param().as_field()
    json_path: fp.StrPath = fp.Param().as_field()
    library_folder: Optional[fp.StrPath] = fp.Param(required=False).as_field()
    library_path: Optional[fp.StrPath] = fp.Param(required=False).as_field()
    default_layers: Mapping[str, Any] = fp.MappingParam(immutable=True).as_field()

    def _default_json_folder(self) -> fp.StrPath:
        return Path(inspect.getfile(type(self))).parent

    def _default_json_name(self) -> fp.StrPath:
        return type(self).__name__

    def _default_json_path(self) -> fp.StrPath:
        return (Path(self.json_folder) / self.json_name).with_suffix(".json")

    def _default_library_path(self) -> Optional[fp.StrPath]:
        library_folder = self.library_folder
        if library_folder is None:
            return None
        json_path = Path(self.json_path)
        library_path = (json_path.parent / library_folder / json_path.name).with_suffix(".gds")
        return library_path

    def _default_default_layers(self) -> Mapping[str, Any]:
        return {"*": "_AUTO_"}

    def build(self):
        insts, elems, ports = super().build()
        cell = fp.import_from_json(name=self.name, json_path=self.json_path, library_path=self.library_path, default_layers=self.default_layers).cell
        for it in cell.content:
            if isinstance(it, fp.ICellRef):  # instance
                insts += it
            else:  # element
                elems += it
        ports += cell.ports
        return insts, elems, ports
